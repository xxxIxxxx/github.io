# 创建一个模块 utils/appConfig.js
```
const appConfig = {
  host: 'https://www.baidu.com',
  sayHi(e) {
    console.log('hi' + e);
  }
}
export default appConfig
```

# 导入创建的模块使用     其他 .js 文件内
```
import appConfig from '../../../utils/appConfig.js';
console.log(appConfig.host);
appConfig.sayHi('朱小明')
```
