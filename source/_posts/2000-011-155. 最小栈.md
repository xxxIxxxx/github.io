# [155. 最小栈](https://leetcode-cn.com/problems/min-stack/)

![155. 最小栈](https://files.catbox.moe/g135og.png)

# 解法

```
class MinStack {
    /** initialize your data structure here. */
    var list: [Int] = []
    //记录当前最小值
    var minNum: Int?
    //只存最小值
    var minNums: [Int] = []

    init() {}

    func push(_ x: Int) {
        list.append(x)
        minNum = min(minNum ?? .max, x)
        if minNum == x {
            minNums.append(minNum!)
        }
    }

    func pop() {
        let p = list.removeLast()
        if minNum == p {
            minNum = list.min()
            minNums.removeLast()
        }
    }

    func top() -> Int {
        return list.last ?? 0
    }

    func getMin() -> Int {
        return minNums.last ?? 0
    }
}

```
